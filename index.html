import io
; The following class shows closures and operator overloading:
dec integer = (num) ->
{
	dec x = int(num)
	return #{
		- -> -x
		+ -> +x
		~ -> ~x
		- y -> x - y
		+ y -> x + y
		* y -> x * y
		% y -> x % y
		/ y -> x / y
		^ y -> x ^ y
		~ y -> x ~ y
		& y -> x & y
		| y -> x | y
		<< y -> x << y
		>> y -> x >> y
	}
}
dec twenty = integer(20)
io.print('20 + 50 = #{twenty + 50}') ; 70
